Timur_Vassalise = {
	if = {
		limit = {
			NOT = { exists = $tag$ }
			any_owned_province = { is_core = $tag$ }
		}
		
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = $tag$
				}
				remove_core = PREV
				add_claim = PREV
				cede_province = $tag$
			}
			#$tag$ = {
			#	Reset_Religion_to_Capital = yes
			#	clr_country_flag = POP_Init
			#	country_event = { id = Init.2 }
			#}
		}
		create_subject = {
			subject_type = $subject_type$
			subject = $tag$
		}
	}
	else = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = $tag$
				}
				remove_core = PREV
				add_claim = PREV
				cede_province = $tag$
			}
		}
	}
}
